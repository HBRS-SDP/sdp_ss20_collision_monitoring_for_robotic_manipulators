193  return false;

195 
196  // For all the link objects (nFrames-1) update the pose.

208 std::vector<double> KinovaArm::ikVelocitySolver(KDL::Twist twist){
209 
210  // vector to store output values
211  std::vector<double> jointVelocitiesOut;
212 
213  // inverse kinemetics solver
214  KDL::ChainIkSolverVel_wdls ikSolver = KDL::ChainIkSolverVel_wdls(fkChain);
215 
216  // Set the weights for singularity handling
217  Eigen::MatrixXd weight_ts;
218  weight_ts.resize(6, 6);
219  weight_ts.setIdentity();
220  weight_ts(0, 0) = 1;
221  weight_ts(1, 1) = 1;
222  weight_ts(2, 2) = 1;
223  weight_ts(3, 3) = 0.4;
224  weight_ts(4, 4) = 0.4;
225  weight_ts(5, 5) = 0.4;
226  ikSolver.setWeightTS(weight_ts);
227 
228  //solver for joint velocities
229  ikSolver.CartToJnt(jointArray, twist, jointVels);
230 
231  // push velocities onto the vector
232  for (int i=0; i<jointVels.rows(); i++){
233  double velocity;
234 
235  // Keep the max velocity to the limit
236  if(jointVels(i) > MAX_JOINT_VEL) {
237  velocity = MAX_JOINT_VEL;
238  }
239  else if(jointVels(i) < -MAX_JOINT_VEL) {
240  velocity = -MAX_JOINT_VEL;
241  }
242  else {
243  velocity = jointVels(i);
244  }
245 
246  // Add the joint angles to the list
247  jointVelocitiesOut.push_back(velocity);
248  }
249 
250  // Return the joint velocities
251  return jointVelocitiesOut;
252 }

294 Eigen::Matrix4d KinovaArm::linkFramesToPose(KDL::Frame startLink, KDL::Frame endLink)
295 {
296  // Convert the frames to eigen matrices for the maths
297  Eigen::Matrix4d startPose = frameToMatrix(startLink);
298  Eigen::Matrix4d endPose = frameToMatrix(endLink);
299 
300  // Find the start and end points
301  Eigen::Vector4d basePoint = startPose * origin;
302  Eigen::Vector4d endPoint = endPose * origin;
303 
304  // create matrix to store the final pose to be returned
305  Eigen::Matrix4d finalPose;
306 
307  // Check to see if the link a starts and originates at the same point
308  if(fabs((basePoint - endPoint).norm()) > 0.0001) {
309  // Get the vector representing the line from the start to end point
310  Eigen::Vector3d midLine = (endPoint - basePoint).head(3);
311  midLine = midLine.normalized();
312  // Get the vector that is prependicular to the midline and z vector
313  // matrix to store the rotaion matrix
314  Eigen::Matrix3d r = i3;
315  // Check to see if the midline is only in the z direction
316 
317  if(midLine(2) != 0) {
318  r << 1, 0, midLine(0),
319  0, 1/midLine(2), midLine(1),
320  0, 0, midLine(2);
321  }
322 
323  // Construct the final matrix based off the start point and rotation Mat
324  finalPose << r(0,0), r(0,1), r(0,2), basePoint(0),
325  r(1,0), r(1,1), r(1,2), basePoint(1),
326  r(2,0), r(2,1), r(2,2), basePoint(2),
327  0, 0, 0, 1;
328  }
329  else {
330  // If they are too close together return a 0 matrix
331  finalPose << 0, 0, 0, 0,
332  0, 0, 0, 0,
333  0, 0, 0, 0,
334  0, 0, 0, 1;
335  }
336 
337  // Return the final pose
338  return finalPose;
339 }
