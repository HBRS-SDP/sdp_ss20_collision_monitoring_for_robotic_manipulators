22  + baseTransform(1,1)
23  + baseTransform(2, 2))/2;
24  double x = (baseTransform(2,1) - baseTransform(1, 2))/(4*w);
25  double y = (baseTransform(0,2) - baseTransform(2, 0))/(4*w);

31  static_transformStamped.transform.translation.y = baseTransform(1,3);
32  static_transformStamped.transform.translation.z = baseTransform(2,3);
33  tf2::Quaternion quat(x, y, z, w);
34  static_transformStamped.transform.rotation.x = quat.x();
35  static_transformStamped.transform.rotation.y = quat.y();
